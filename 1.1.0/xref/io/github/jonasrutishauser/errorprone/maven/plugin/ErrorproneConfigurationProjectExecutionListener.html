3   import javax.inject.Inject;
4   import javax.inject.Named;
5   import javax.inject.Provider;
6   
7   import org.apache.maven.execution.ProjectExecutionEvent;
8   import org.apache.maven.execution.ProjectExecutionListener;
9   import org.apache.maven.lifecycle.LifecycleExecutionException;
10  import org.apache.maven.plugin.MojoExecution;
11  
12  @Named
13  public class ErrorproneConfigurationProjectExecutionListener implements ProjectExecutionListener {
14  
15      private final Provider<CompilerConfiguration> compilerConfiguration;
16  
17      @Inject
18      ErrorproneConfigurationProjectExecutionListener(Provider<CompilerConfiguration> compilerConfiguration) {
19          this.compilerConfiguration = compilerConfiguration;
20      }
21  
22      @Override
23      public void beforeProjectExecution(ProjectExecutionEvent event) throws LifecycleExecutionException {
24          // nothing to do
25      }
26  
27      @Override
28      public void beforeProjectLifecycleExecution(ProjectExecutionEvent event) throws LifecycleExecutionException {
29          CompilerConfiguration configuration = compilerConfiguration.get();
30          configuration.setCompilerExecutions(event.getProject(), event.getExecutionPlan().stream()
31                  .filter(execution -> isJavacCompilerExecution(configuration, execution)).toList());
32      }
33  
34      private boolean isJavacCompilerExecution(CompilerConfiguration configuration, MojoExecution mojoExecution) {
35          if ("org.apache.maven.plugins".equals(mojoExecution.getGroupId())
36                  && "maven-compiler-plugin".equals(mojoExecution.getArtifactId())
37                  && ("compile".equals(mojoExecution.getGoal()) || "testCompile".equals(mojoExecution.getGoal()))) {
38              String value = configuration.getParameterValue(mojoExecution, mojoExecution.getConfiguration().getChild("compilerId"));
39              return value == null || "javac".equals(value);
40          }
41          return false;
42      }
43  
44      @Override
45      public void afterProjectExecutionSuccess(ProjectExecutionEvent event) throws LifecycleExecutionException {
46          compilerConfiguration.get().clearCompilerExecutions(event.getProject());
47      }
48  
49      @Override
50      public void afterProjectExecutionFailure(ProjectExecutionEvent event) {
51          compilerConfiguration.get().clearCompilerExecutions(event.getProject());
52      }

54  }
